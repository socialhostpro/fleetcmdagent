## API INTEGRATION

```typescript
// api/hooks.ts - React Query hooks

// Nodes
export const useNodes = () => useQuery(['nodes'], fetchNodes);
export const useNode = (id: string) => useQuery(['node', id], () => fetchNode(id));
export const useNodeMetrics = (id: string) => useQuery(['metrics', id], () => fetchMetrics(id), { refetchInterval: 5000 });

// Clusters
export const useClusters = () => useQuery(['clusters'], fetchClusters);
export const useClusterJobs = (cluster: string) => useQuery(['jobs', cluster], () => fetchClusterJobs(cluster));

// Docker
export const useServices = () => useQuery(['services'], fetchServices);
export const useContainers = (nodeId: string) => useQuery(['containers', nodeId], () => fetchContainers(nodeId));

// Jobs
export const useJobs = () => useQuery(['jobs'], fetchJobs, { refetchInterval: 2000 });
export const useCreateJob = () => useMutation(createJob);

// Roamers
export const useRoamers = () => useQuery(['roamers'], fetchRoamers);
export const useAssignRoamer = () => useMutation(assignRoamer);

// WebSocket for real-time updates
export const useMetricsStream = () => {
  const ws = useWebSocket('ws://spark:8765/ws/metrics');
  return ws.data;
};
```

## SUMMARY

Complete UI specification including:

1. **Main Views:** Canvas, Clusters, Network, Docker, Storage, Backup, Metrics, Robotics
2. **Canvas Features:** n8n-style nodes, animated edges, traffic flow, zoom/pan
3. **Node Types:** Spark, Xavier, Roamer (with status indicators)
4. **Panels:** Detail (right), Bottom (terminals/logs/jobs)
5. **Interactions:** Context menus, keyboard shortcuts, drag-drop
6. **Pop-outs:** Separate terminal windows
7. **Modals:** Assign roamer, deploy service, backup, etc.
8. **Tech Stack:** React, TypeScript, React Flow, D3, xterm.js, Tailwind
